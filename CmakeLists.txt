cmake_minimum_required(VERSION 3.10)
project(CmakeCpp)
include_directories(include)

# Create a static library from utils.cpp
add_library(utils STATIC src/utils.cpp)

# Define source files for the executable
set(SOURCE_FILES src/main.cpp src/ThirdAction.cpp)
add_executable(hello ${SOURCE_FILES})

# Link the executable to the utils static library
target_link_libraries(hello utils)
